1. Binary Gap

Find the longest sequence of zeros in the binary representation of an integer that is surrounded by ones at both ends.
                    function solution(N) {
                        let binary = N.toString(2);
                        let maxGap = 0;
                        let currentGap = 0;
                        let counting = false;

                        for ( let bit of binary ) {
                            if (bit === '1') {
                                if (counting) {
                                    maxGap = Math.max(maxGap, currentGap);
                                }
                                counting = true;
                                currentGap = 0;
                            } else if( counting ) {
                                currentGap++;
                            }
                        }

                        return maxGap;
                    }

2. Cyclic Rotation

Rotate an array K times to the right.
                    function solution(A, K) {
                        const N = A.length;
                        if (N === 0) return A;

                        const rotations = K % N;
                        if (rotations === 0) return A;

                        return A.slice(-rotations).concat(A.slice(0, N - rotations));
                    }

2.1 Odd Occurrences in Array

Find the value that occurs in odd number of elements in the array.
                    function solution(A) {
                        const count = {};
                        for (let i = 0; i < A.length; i++) {
                            count[A[i]] = (count[A[i]] || 0) + 1;
                        }
                        for (const key in count) {
                            if (count[key] % 2 === 1) {
                                return parseInt(key);
                            }
                        }
                        return 0;
                    }

                    // alternative solution using XOR
                    function solutionXOR(A) {
                        let result = 0;
                        for (let num of A) {
                            result ^= num; 
                        }
                        return result;
                    }
